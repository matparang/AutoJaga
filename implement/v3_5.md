# jagabot/lab/scaling.py

import asyncio
from typing import List, Dict, Any, Optional
from dataclasses import dataclass
from datetime import datetime, timedelta
import psutil

@dataclass
class ScalingConfig:
    """Configuration for auto-scaling behavior"""
    min_workers: int = 2
    max_workers: int = 32
    scale_up_threshold: int = 5  # Queue length to trigger scale up
    scale_down_threshold: int = 2  # Queue length to trigger scale down
    cooldown_period: int = 60  # Seconds between scaling events
    target_cpu_usage: float = 70.0  # Target CPU percentage
    scale_up_factor: float = 1.5  # Multiply workers by this when scaling up
    scale_down_factor: float = 0.5  # Reduce workers by this when scaling down

class ScalableWorkerPool:
    """
    Worker pool that automatically scales based on demand
    """
    
    def __init__(self, config: Optional[ScalingConfig] = None):
        self.config = config or ScalingConfig()
        self.workers = []
        self.task_queue = asyncio.Queue()
        self.active_tasks = {}
        self.scaling_lock = asyncio.Lock()
        self.last_scale_time = datetime.now()
        self.metrics = {
            'scale_up_events': 0,
            'scale_down_events': 0,
            'peak_workers': 0,
            'total_tasks_processed': 0
        }
        
        # Start with minimum workers
        self.current_workers = self.config.min_workers
        self._start_workers(self.current_workers)
        
        # Start monitoring task
        asyncio.create_task(self._monitor_loop())
    
    async def submit_task(self, tool: str, params: dict, priority: int = 5) -> str:
        """
        Submit task to pool with priority
        """
        task_id = f"task_{datetime.now().strftime('%Y%m%d_%H%M%S')}_{len(self.active_tasks)}"
        
        await self.task_queue.put({
            'id': task_id,
            'tool': tool,
            'params': params,
            'priority': priority,
            'submitted': datetime.now()
        })
        
        return task_id
    
    async def _monitor_loop(self):
        """
        Monitor queue and system metrics to trigger scaling
        """
        while True:
            await asyncio.sleep(5)  # Check every 5 seconds
            
            queue_size = self.task_queue.qsize()
            cpu_percent = psutil.cpu_percent()
            
            async with self.scaling_lock:
                await self._evaluate_scaling_needs(queue_size, cpu_percent)
    
    async def _evaluate_scaling_needs(self, queue_size: int, cpu_percent: float):
        """
        Decide whether to scale up or down
        """
        now = datetime.now()
        seconds_since_last_scale = (now - self.last_scale_time).total_seconds()
        
        if seconds_since_last_scale < self.config.cooldown_period:
            return  # Still in cooldown
        
        # Scale up conditions
        if (queue_size > self.config.scale_up_threshold and 
            self.current_workers < self.config.max_workers):
            
            new_worker_count = min(
                int(self.current_workers * self.config.scale_up_factor),
                self.config.max_workers
            )
            await self._scale_to(new_worker_count)
            self.metrics['scale_up_events'] += 1
        
        # Scale down conditions
        elif (queue_size < self.config.scale_down_threshold and 
              self.current_workers > self.config.min_workers and
              cpu_percent < self.config.target_cpu_usage):
            
            new_worker_count = max(
                int(self.current_workers * self.config.scale_down_factor),
                self.config.min_workers
            )
            await self._scale_to(new_worker_count)
            self.metrics['scale_down_events'] += 1
    
    async def _scale_to(self, new_count: int):
        """
        Scale worker pool to new count
        """
        if new_count == self.current_workers:
            return
        
        print(f"🔄 Scaling workers: {self.current_workers} → {new_count}")
        
        if new_count > self.current_workers:
            # Scale up - add workers
            self._start_workers(new_count - self.current_workers)
        else:
            # Scale down - gracefully remove workers
            await self._stop_workers(self.current_workers - new_count)
        
        self.current_workers = new_count
        self.last_scale_time = datetime.now()
        self.metrics['peak_workers'] = max(self.metrics['peak_workers'], new_count)
    
    def _start_workers(self, count: int):
        """Start new worker tasks"""
        for i in range(count):
            worker_id = len(self.workers)
            worker_task = asyncio.create_task(self._worker_loop(worker_id))
            self.workers.append(worker_task)
    
    async def _stop_workers(self, count: int):
        """Gracefully stop workers (let them finish current task)"""
        # Implementation: set stop flag, workers exit after current task
        pass
    
    async def _worker_loop(self, worker_id: int):
        """
        Main worker loop - processes tasks from queue
        """
        while True:
            try:
                # Get task with timeout (to allow scaling down)
                task = await asyncio.wait_for(
                    self.task_queue.get(), 
                    timeout=5.0
                )
                
                # Process task
                start = datetime.now()
                result = await self._execute_task(task)
                duration = (datetime.now() - start).total_seconds()
                
                self.metrics['total_tasks_processed'] += 1
                
                # Store result
                self.active_tasks[task['id']] = {
                    'result': result,
                    'duration': duration,
                    'worker': worker_id
                }
                
                self.task_queue.task_done()
                
            except asyncio.TimeoutError:
                # No tasks for 5 seconds - check if we should exit
                if self.current_workers > self.config.min_workers:
                    # This worker might be scaled down
                    continue

from jagabot.lab.scaling import ScalableWorkerPool, ScalingConfig

class ParallelLab:
    """
    ParallelLab with auto-scaling worker pool
    """
    
    def __init__(self, 
                 min_workers: int = 2,
                 max_workers: int = 32,
                 scale_up_threshold: int = 5,
                 scale_down_threshold: int = 2):
        
        config = ScalingConfig(
            min_workers=min_workers,
            max_workers=max_workers,
            scale_up_threshold=scale_up_threshold,
            scale_down_threshold=scale_down_threshold
        )
        
        self.pool = ScalableWorkerPool(config)
        self.workflows = self._init_workflows()
    
    async def submit_tasks(self, tasks: List[Dict]) -> List[str]:
        """
        Submit multiple tasks with auto-scaling
        """
        task_ids = []
        for task in tasks:
            task_id = await self.pool.submit_task(
                task['tool'],
                task['params'],
                task.get('priority', 5)
            )
            task_ids.append(task_id)
        return task_ids
    
    async def get_metrics(self) -> Dict:
        """Get scaling metrics"""
        return self.pool.metrics

import pytest

async def test_scale_up_on_queue_growth():
    config = ScalingConfig(

    config = ScalingConfig(
        min_workers=2,
        max_workers=8,
        scale_down_threshold=1,
        cooldown_period=1
    )
    
    pool = ScalableWorkerPool(config)

    await asyncio.sleep(5)
    
    assert pool.current_workers == config.min_workers
