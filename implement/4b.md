```python
# jagabot/evolution/engine.py

class EvolutionEngine:
    """Creates new tools at runtime"""
    
    def __init__(self):
        self.memory = MemoryFleet()
        self.meta = MetaLearningEngine()
        self.templates = self.load_templates()
        self.registry = ToolRegistry()
    
    def create_tool(self, specification):
        """Main entry point - create new tool from spec"""
        
        # Step 1: Check if similar tool already exists
        existing = self.find_similar(specification)
        if existing:
            return {'existing': existing, 'message': 'Tool already exists'}
        
        # Step 2: Generate code
        code = self.generate_code(specification)
        
        # Step 3: Test in sandbox
        test_result = self.test_in_sandbox(code, specification.test_cases)
        
        if not test_result['passed']:
            # Learn from failure
            self.meta.track_failed_creation(specification, test_result)
            return {'error': 'Tests failed', 'details': test_result}
        
        # Step 4: Register tool
        tool_id = self.registry.register(
            name=specification.name,
            code=code,
            description=specification.description,
            parameters=specification.parameters
        )
        
        # Step 5: Update documentation
        self.update_skill_md(specification)
        
        # Step 6: Track in memory
        self.memory.store_tool_creation(tool_id, specification)
        
        return {
            'success': True,
            'tool_id': tool_id,
            'name': specification.name,
            'message': f'New tool {specification.name} created and registered'
        }
    
    def generate_code(self, spec):
        """Generate Python code from template"""
        template = self.templates[spec.template_type]
        
        # Fill template with spec
        code = template.replace('{{TOOL_NAME}}', spec.name)
        code = code.replace('{{DESCRIPTION}}', spec.description)
        code = code.replace('{{PARAMETERS}}', json.dumps(spec.parameters))
        code = code.replace('{{LOGIC}}', spec.logic)
        
        return code
```

```python
# jagabot/evolution/spec_parser.py

class ToolSpecification:
    """User request → tool specification"""
    
    @classmethod
    def from_user_query(cls, query):
        """Parse user query into tool spec"""
        
        # Use LLM to extract:
        # - What type of tool needed
        # - Input parameters
        # - Expected output
        # - Test cases
        
        spec = cls()
        
        if 'correlation' in query.lower():
            spec.template_type = 'correlation'
            spec.name = generate_tool_name(query)
            spec.description = query
            
            if 'bitcoin' in query.lower() and 'oil' in query.lower():
                spec.test_cases = [
                    {
                        'input': {'asset1': 'BTC-USD', 'asset2': 'CL=F'},
                        'expected': {'correlation': 'float'}
                    }
                ]
        
        return spec
```

4. Sandbox Tester

```python
# jagabot/evolution/tester.py

class ToolTester:
    """Test generated tools in sandbox"""
    
    def __init__(self):
        self.sandbox = SafePythonExecutor()
    
    def test_tool(self, code, test_cases):
        """Run test cases against generated tool"""
        
        results = []
        for case in test_cases:
            # Wrap code with test
            test_code = f'''
{code}

# Test
tool = {{TOOL_NAME}}Tool()
result = tool.execute({case['input']})
print(result)
'''
            output = self.sandbox.execute(test_code)
            
            passed = self.validate_output(output, case['expected'])
            results.append({
                'case': case,
                'passed': passed,
                'output': output
            })
        
        return {
            'passed': all(r['passed'] for r in results),
            'results': results
        }
```

```python
@cli.group()
def evolution():

@evolution.command()
@click.argument('query')
def create(query):
    """Create new tool from description"""
    engine = EvolutionEngine()
    spec = ToolSpecification.from_user_query(query)
    result = engine.create_tool(spec)
    
    if result.get('success'):
        click.echo(f"✅ New tool created: {result['name']}")
        click.echo(f"Tool ID: {result['tool_id']}")
    else:
        click.echo(f"❌ Failed: {result.get('error', 'Unknown error')}")

@evolution.command()
def list():
    """List all evolved tools"""
    registry = ToolRegistry()
    tools = registry.list_evolved()
    
    table = Table(title="Evolved Tools")
    table.add_column("ID", style="cyan")
    table.add_column("Name", style="green")
    table.add_column("Created", style="yellow")
    table.add_column("Usage", style="magenta")
    
    for t in tools:
        table.add_row(t.id, t.name, t.created, str(t.usage_count))
    
    console.print(table)

@evolution.command()
@click.argument('tool_id')
def test(tool_id):
    """Test an evolved tool"""
    tester = ToolTester()
    result = tester.test_tool_by_id(tool_id)
    
    if result['passed']:
        click.echo(f"✅ Tool {tool_id} passed all tests")
    else:
        click.echo(f"❌ Tool {tool_id} failed:")
        for r in result['results']:
            if not r['passed']:
                click.echo(f"  - Test failed: {r['case']}")
```
